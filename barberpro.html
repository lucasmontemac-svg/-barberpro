-- Recriar policies (caso já existam, dropa primeiro) drop policy if exists "owner_barbearias" on bp_barbearias; drop policy if exists "owner_barbeiros" on bp_barbeiros; drop policy if exists "owner_servicos" on bp_servicos; drop policy if exists "owner_clientes" on bp_clientes; drop policy if exists "owner_agendamentos" on bp_agendamentos; drop policy if exists "owner_caixa" on bp_caixa; create policy "owner_barbearias" on bp_barbearias for all using (user_id = auth.uid()); create policy "owner_barbeiros" on bp_barbeiros for all using (barbearia_id in (select id from bp_barbearias where user_id = auth.uid())); create policy "owner_servicos" on bp_servicos for all using (barbearia_id in (select id from bp_barbearias where user_id = auth.uid())); create policy "owner_clientes" on bp_clientes for all using (barbearia_id in (select id from bp_barbearias where user_id = auth.uid())); create policy "owner_agendamentos" on bp_agendamentos for all using (barbearia_id in (select id from bp_barbearias where user_id = auth.uid())); create policy "owner_caixa" on bp_caixa for all using (barbearia_id in (select id from bp_barbearias where user_id = auth.uid()));
